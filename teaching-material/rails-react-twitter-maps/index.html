Rails / React - Working With A Third Party API (Twitter) and Google Maps

At some point in an applications development lifecycle the product will need data that isn’t provided by the applications database. This is where 3rd party API’s come in. There are many different 3rd party API’s. Today we will be building a Rails JSON API that serves up data to our React components from the Twitter API. We will also go through a Google Maps example in React.

Create A TwitterClient Helper Class - lib/twitter_client.rb

class TwitterClient
    def self.home_timeline
        client.home_timeline
    end
    def self.tweet(message)
        client.update(message)
    end
    private
      def self.client
        Twitter::REST::Client.new do |config|
            config.consumer_key        = ENV['TWITTER_CONSUMER_KEY']
            config.consumer_secret     = ENV['TWITTER_CONSUMER_SECRET']
            config.access_token        = ENV['TWITTER_ACCESS_TOKEN']
            config.access_token_secret = ENV['TWITTER_ACCESS_TOKEN_SECRET']
        end
      end
end

Tweets Component - client/components/Tweets.js

import React, { Component } from 'react';
import Tweet from './Tweet';
class Tweets extends Component {
    constructor(props) {
        super(props);
        this.state = { tweets: [] };
        this.tweet = this.tweet.bind(this);
    }
    componentDidMount() {
        $.ajax({
            url: '/api/home_timeline',
            type: 'GET',
            dataType: 'JSON'
        }).done( tweets => {
            this.setState({ tweets });
        }).fail(data => {
            console.log(data);
        });
    }
    displayTweets() {
        let tweets = this.state.tweets;
        if(tweets.length) {
            return tweets.map( tweet => {
                return(<Tweet key={tweet.id} tweet={tweet} />);
            });
        } else {
            return(<h4>No Tweets To Display</h4>);
        }
    }
    tweet(e) {
        e.preventDefault();
        $.ajax({
            url: '/api/tweet',
            type: 'POST',
            data: { message: this.refs.message.value },
            dataType: 'JSON'
        }).done( tweet => {
            this.setState({
                tweets: [
                  tweet,
                  ...this.state.tweets
                ]
          });
            this.refs.tweetForm.reset();
        }).fail(data => {
            console.log(data);
        });
    }
    render() {
        return(
            <div>
              <h3>All My Tweets</h3>
              <hr />
              <form ref='tweetForm' onSubmit={this.tweet}>
                <textarea className='form-control' required ref='message' placeholder="Your Tweet Here!"></textarea>
                <button className='btn btn-success'>Send Tweet</button>
              </form>
              <hr />
              { this.displayTweets() }
            </div>
        );
    }
}
export default Tweets;

GoogleMap Component - client/components/GoogleMap.js

import React, { Component } from 'react';
class GoogleMap extends Component {
    componentDidMount() {
        let handler = Gmaps.build('Google');
        let markers;
        handler.buildMap({ provider: { }, internal: {id: 'map'}}, () => {
          markers = handler.addMarkers([
            {
              "lat": 40.7609915,
              "lng": -111.88287990000003,
              "picture": {
                "url": "https://camo.githubusercontent.com/fda67f6a65bb639a73cc56a5f678964c89f7d672/687474703a2f2f6465762e626f7764656e7765622e636f6d2f612f692f6a732f69636f6e732f6a6176617363726970742d69636f6e2d33322e706e67",
                "width":  32,
                "height": 32
              },
              "infowindow": "DevPoint Labs Is Awesome!"
            }
          ]);
          handler.bounds.extendWith(markers);
          handler.fitMapToBounds();
          handler.getMap().setZoom(18);
        });
    }
    render() {
        return(
            <div style={styles.mapContainer}>
              <div id="map" style={styles.map}></div>
            </div>
        );
    }
}
const styles = {
    mapContainer: {
        width: '800px'
    },
    map: {
        width: '800px',
        height: '400px'
    }
}
export default GoogleMap;
